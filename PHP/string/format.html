• $name = trim/ltrim/rtrim($_POST('name') [, список_удаляемых_символов]); - усечение
  • echo nl2br($mailcontent); - замена переводов строки на <br />
  • string sprintf(string формат, [, mixed аргументы ...]); - возврат отформатированной строки
  • void printf(string формат, [, mixed аргументы ...]); - вывод отформатированной строки в браузер
    • формат : %[ номер_аргумента\$ ][ 'дополняющий_символ ][ - ][ ширина ][ .точность ] тип (если надо вывести %, то %%)
    • [ дополняющий символ ] - для дополнения до указанной ширины (пробел по умолчанию), если 0 или пробел, то кавычка не нужна
    • [ - ] выравнивание по левому краю
    • [ ширина ] в символах для переменной
    • тип :
      • b - целое в двоичное число
      • c - целое в симол
      • d - целое в десятичное число
      • f - двойной точности в плавающую точку
      • o - целое в восьмеричное
      • s - строка
      • u - целое десятичное без знака
      • x - целое в 16-е строчными
      • X - целое в 16-е прописными
  • vprintf(), vsprintf() - то же но второй параметр - массив аргументов
  • strtoupper() - в прописные
  • strtolower() - в строчные
  • ucfirst() - первый символ в прописные
  • ucwords() - первый символ каждого слова в прописные
  • $feedback = addslash($str) - добавление слешей перед спец.символами (сервер сам может добавлять если установлен magic_quotes_gpc - можно проверит функцией get_magic_quotes_gpc())
  • stripslash() - удаление слешей (кроме двойных)

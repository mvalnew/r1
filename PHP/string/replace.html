• string strstr / strchar(string где_поиск, string что_ищем) - возвращает часть строки начиная с искомой или false
  • strrchar() - возвращает подстроку начиная с последнего вхождения искомой
  • stristr() - нечувствительна к регистру
  • int strpos(string где_поиск, string что_ищем, [ int смещение ]) - возвращает позицию искомой строки или false
  • strrpos() - последнее вхождение
  • mixed str_replace(mixed что_ищем, mixed чем_заменяем, mixed где_заменяем [, int количество_замен]) - поиск и замена (параметрами могут быть массивы строк)
  • string substr_replace(string где_меняем, string чем_заменяем, int начало_замены [, int длина_замены]) - замена в указанной позиции (если начало_замены < 0, то отсчет с конца, если длина замены < 0, то это позиция с конца строки, где замена прекращается)
